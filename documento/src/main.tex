\ifXeTeX
  \usepackage{fontspec}

  \IfFontExistsTF{Aptos}{%
    \setmainfont{Aptos}%
  }{%
    \IfFontExistsTF{Calibri}{%
      \setmainfont{Calibri}%
    }{%

      }{%
        \usepackage{lmodern}%

    }%

% ---------- JERARQUÍA DE TÍTULOS (APA 7, 5 niveles) ----------
% Nivel 1 – centrado, negrita
% Nivel 2 – alineado izquierda, negrita
% Nivel 3 – alineado izquierda, negrita cursiva
% Nivel 4 – con sangría, negrita, termina en punto
% Nivel 5 – con sangría, negrita cursiva, termina en punto
\usepackage{titlesec}

% Sin numeración
\setcounter{secnumdepth}{0}

% Nivel 1: \section  – centrado, negrita, 12pt
\titleformat{\section}
  {\normalfont\bfseries\normalsize\centering}
  {}{0pt}{}
\titlespacing*{\section}{0pt}{12pt}{6pt}

% Nivel 2: \subsection – izquierda, negrita
\titleformat{\subsection}
  {\normalfont\bfseries\normalsize\raggedright}
  {}{0pt}{}
\titlespacing*{\subsection}{0pt}{12pt}{6pt}

% Nivel 3: \subsubsection – izquierda, cursiva sin negrita
\titleformat{\subsubsection}
  {\normalfont\itshape\normalsize\raggedright}
  {}{0pt}{}
\titlespacing*{\subsubsection}{0pt}{12pt}{6pt}
